### Task P3-1: 实现 L5 交叉验证

**Files:**
- Modify: `app/src/main/kotlin/dev/minios/ocremote/data/repository/SessionStatusManager.kt`

- [ ] **Step 1: 在 SessionStatusManager 中添加 L5 检查**

在 `init` 块中添加 L5 检查的启动，或在 `checkStaleness` 中合并：

```kotlin
private fun checkStaleness() {
    val now = System.currentTimeMillis()
    val messages = eventDispatcher.messages.value
    
    _fsmStates.value.forEach { (sessionId, state) ->
        // L2: Busy 但 stale
        if (state.core is SessionStatus.Busy) {
            val staleFor = now - state.lastEventAt
            if (staleFor > STALENESS_THRESHOLD_MS) {
                Log.w(TAG, "[$sessionId] L2 stale for ${staleFor}ms")
                triggerRestValidation(sessionId)
            }
        }
        
        // L5: Idle 但有 incomplete assistant message
        if (state.core is SessionStatus.Idle) {
            val hasIncomplete = messages[sessionId]
                ?.filterIsInstance<dev.minios.ocremote.domain.model.Message.Assistant>()
                ?.any { it.time.completed == null }
                ?: false
            if (hasIncomplete) {
                Log.w(TAG, "[$sessionId] L5 inconsistency: Idle but has incomplete assistant")
                triggerRestValidation(sessionId)
            }
        }
    }
}
```

private fun triggerRestValidation(sessionId: String) {
    val sid = currentServerId ?: return
    appScope.launch {
        try {
            val result = sessionRepository.fetchSessionStatuses(sid, directory = null)
            result.onSuccess { statuses ->
                val serverStatus = statuses[sessionId]
                if (serverStatus != null) {
                    if (BuildConfig.DEBUG) {
                        Log.d(TAG, "[$sessionId] L3 REST validation: server says ${serverStatus::class.simpleName}")

                    onRestValidation(sessionId, serverStatus)

        }
    }
